## Pulling Backtest Results

### Available Backtest Data

When you run a backtest, YouTrade stores comprehensive data including:

- **Performance Metrics**: Win rate, profit/loss, Sharpe ratio, max drawdown
- **Trade Details**: Entry/exit prices, timestamps, profit per trade
- **Time Series Data**: Equity curve, drawdown periods, monthly/weekly breakdown
- **Pair Analysis**: Performance per trading pair
- **Statistical Data**: Distribution of wins/losses, average hold time

### Example Queries

## Pulling Hyperopt Results

### What is Hyperopt Data?

Hyperopt (hyperparameter optimization) tests thousands of parameter combinations to find the best settings for your strategy. The results include:

- **Best Parameters**: Optimal values for each parameter
- **Epoch Data**: Results from each optimization iteration
- **Convergence Analysis**: How the optimization improved over time
- **Parameter Importance**: Which parameters had the most impact
- **Statistical Confidence**: Reliability of the results

### Querying Hyperopt Results

<CodeGroup>

```bash Bash
# Get hyperopt summary
curl -X GET "https://api.youtrade.ai/api/v1/strategies/{strategyId}/hyperopt/{hyperoptId}" \
  -H "Authorization: Bearer $YOUTRADE_API_KEY"

# Get all epochs
curl -X GET "https://api.youtrade.ai/api/v1/strategies/{strategyId}/hyperopt/{hyperoptId}/epochs" \
  -H "Authorization: Bearer $YOUTRADE_API_KEY"

# Get best parameters
curl -X GET "https://api.youtrade.ai/api/v1/strategies/{strategyId}/hyperopt/{hyperoptId}/best-params" \
  -H "Authorization: Bearer $YOUTRADE_API_KEY"
```

```python Python
import requests
import os

API_KEY = os.getenv('YOUTRADE_API_KEY')
BASE_URL = 'https://api.youtrade.ai'

strategy_id = 'your-strategy-id'
hyperopt_id = 'your-hyperopt-id'

# Get hyperopt results
hyperopt_results = requests.get(
    f'{BASE_URL}/api/v1/strategies/{strategy_id}/hyperopt/{hyperopt_id}',
    headers={'Authorization': f'Bearer {API_KEY}'}
).json()

print("=== Hyperopt Results ===\n")
print(f"Total Epochs: {hyperopt_results['total_epochs']}")
print(f"Best Epoch: {hyperopt_results['best_epoch']}")
print(f"Best Profit: {hyperopt_results['best_result']['total_profit']:.2f}%")

print("\n=== Best Parameters ===")
for param, value in hyperopt_results['best_params'].items():
    print(f"{param}: {value}")

# Get parameter importance
print("\n=== Parameter Importance ===")
for param in hyperopt_results['parameter_importance']:
    print(f"{param['name']}: {param['importance']:.1%}")
```

```javascript JavaScript
const API_KEY = process.env.YOUTRADE_API_KEY;
const strategyId = 'your-strategy-id';
const hyperoptId = 'your-hyperopt-id';

async function getHyperoptResults() {
  const response = await fetch(
    `https://api.youtrade.ai/api/v1/strategies/${strategyId}/hyperopt/${hyperoptId}`,
    {
      headers: {
        'Authorization': `Bearer ${API_KEY}`
      }
    }
  );
  
  const results = await response.json();
  
  console.log('=== Hyperopt Results ===\n');
  console.log(`Total Epochs: ${results.total_epochs}`);
  console.log(`Best Epoch: ${results.best_epoch}`);
  console.log(`Best Profit: ${results.best_result.total_profit.toFixed(2)}%`);
  
  console.log('\n=== Best Parameters ===');
  Object.entries(results.best_params).forEach(([param, value]) => {
    console.log(`${param}: ${value}`);
  });
  
  return results;
}

getHyperoptResults();
```

## Pulling Dry-Run Data

### What is Dry-Run Data?

Dry-run (paper trading) lets you test strategies with real market data but simulated money. This data is crucial for validating strategies before risking real capital.

### Accessing Dry-Run Data

<CodeGroup>

```bash Bash
# Get current dry-run status
curl -X GET "https://api.youtrade.ai/api/v1/deployments/{deploymentId}/status" \
  -H "Authorization: Bearer $YOUTRADE_API_KEY"

# Get dry-run trades
curl -X GET "https://api.youtrade.ai/api/v1/deployments/{deploymentId}/trades?mode=dry_run" \
  -H "Authorization: Bearer $YOUTRADE_API_KEY"

# Get dry-run performance metrics
curl -X GET "https://api.youtrade.ai/api/v1/deployments/{deploymentId}/metrics?mode=dry_run" \
  -H "Authorization: Bearer $YOUTRADE_API_KEY"
```

```python Python
import requests
import os
from datetime import datetime, timedelta

API_KEY = os.getenv('YOUTRADE_API_KEY')
BASE_URL = 'https://api.youtrade.ai'

deployment_id = 'your-deployment-id'

# Get current status
status = requests.get(
    f'{BASE_URL}/api/v1/deployments/{deployment_id}/status',
    headers={'Authorization': f'Bearer {API_KEY}'}
).json()

print(f"Strategy: {status['strategy_name']}")
print(f"Mode: {status['mode']}")
print(f"Status: {status['status']}")
print(f"Uptime: {status['uptime']}")

# Get trades from last 7 days
trades = requests.get(
    f'{BASE_URL}/api/v1/deployments/{deployment_id}/trades',
    headers={'Authorization': f'Bearer {API_KEY}'},
    params={
        'mode': 'dry_run',
        'start_date': (datetime.now() - timedelta(days=7)).isoformat(),
        'end_date': datetime.now().isoformat()
    }
).json()

print(f"\n=== Last 7 Days Performance ===")
print(f"Total Trades: {len(trades['trades'])}")
print(f"Winning Trades: {sum(1 for t in trades['trades'] if t['profit'] > 0)}")

# Calculate metrics
total_profit = sum(t['profit_percent'] for t in trades['trades'])
print(f"Total Profit: {total_profit:.2f}%")
```

```javascript JavaScript
const API_KEY = process.env.YOUTRADE_API_KEY;
const deploymentId = 'your-deployment-id';

async function getDryRunData() {
  // Get current status
  const statusRes = await fetch(
    `https://api.youtrade.ai/api/v1/deployments/${deploymentId}/status`,
    {
      headers: { 'Authorization': `Bearer ${API_KEY}` }
    }
  );
  const status = await statusRes.json();
  
  console.log(`Strategy: ${status.strategy_name}`);
  console.log(`Mode: ${status.mode}`);
  console.log(`Status: ${status.status}`);
  
  // Get trades
  const tradesRes = await fetch(
    `https://api.youtrade.ai/api/v1/deployments/${deploymentId}/trades?mode=dry_run`,
    {
      headers: { 'Authorization': `Bearer ${API_KEY}` }
    }
  );
  const trades = await tradesRes.json();
  
  console.log(`\nTotal Trades: ${trades.trades.length}`);
  
  const winningTrades = trades.trades.filter(t => t.profit > 0);
  console.log(`Win Rate: ${(winningTrades.length / trades.trades.length * 100).toFixed(1)}%`);
  
  return { status, trades };
}

getDryRunData();
```

## Advanced Reporting Queries

### Custom Time Range Analysis

Get data for specific time periods:

```python
import requests
from datetime import datetime, timedelta

# Get trades from last 30 days
start_date = (datetime.now() - timedelta(days=30)).isoformat()
end_date = datetime.now().isoformat()

trades = requests.get(
    f'{BASE_URL}/api/v1/deployments/{deployment_id}/trades',
    headers={'Authorization': f'Bearer {API_KEY}'},
    params={
        'start_date': start_date,
        'end_date': end_date
    }
).json()

# Analyze by week
weekly_stats = {}
for trade in trades['trades']:
    week = datetime.fromisoformat(trade['close_date']).strftime('%Y-W%W')
    if week not in weekly_stats:
        weekly_stats[week] = {'trades': 0, 'profit': 0, 'wins': 0}
    
    weekly_stats[week]['trades'] += 1
    weekly_stats[week]['profit'] += trade['profit_percent']
    if trade['profit_percent'] > 0:
        weekly_stats[week]['wins'] += 1

for week, stats in sorted(weekly_stats.items()):
    win_rate = (stats['wins'] / stats['trades'] * 100) if stats['trades'] > 0 else 0
    print(f"{week}: {stats['trades']} trades, {stats['profit']:.2f}% profit, {win_rate:.1f}% win rate")
```

### Export Data for External Analysis

Export data to CSV or JSON for analysis in Excel, Python, or R:

```python
import csv
import json

def export_backtest_trades(strategy_id: str, backtest_id: str, format: str = 'csv'):
    """Export backtest trades to file"""
    
    trades = requests.get(
        f'{BASE_URL}/api/v1/strategies/{strategy_id}/backtest/{backtest_id}/trades',
        headers={'Authorization': f'Bearer {API_KEY}'}
    ).json()
    
    if format == 'csv':
        with open(f'backtest_{backtest_id}.csv', 'w', newline='') as f:
            writer = csv.DictWriter(f, fieldnames=trades['trades'][0].keys())
            writer.writeheader()
            writer.writerows(trades['trades'])
        print(f"Exported to backtest_{backtest_id}.csv")
    
    elif format == 'json':
        with open(f'backtest_{backtest_id}.json', 'w') as f:
            json.dump(trades, f, indent=2)
        print(f"Exported to backtest_{backtest_id}.json")

# Export to CSV
export_backtest_trades('str_abc123', 'bt_xyz789', format='csv')
```
